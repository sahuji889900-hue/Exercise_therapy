function shuffle(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
}

  html += `<div class="answer"></div></div>`;
  quiz.innerHTML += html;

/* ===============================
   INSTANT FEEDBACK
=============================== */
function check(element, isCorrect) {
  const options = element.parentElement.querySelectorAll('.option');
  const answerBox = element.parentElement.querySelector('.answer');

  options.forEach(opt => opt.onclick = null);

  if (isCorrect) {
    element.classList.add('correct');
    answerBox.innerHTML = "✅ Correct";
  } else {
    element.classList.add('wrong');
    answerBox.innerHTML = "❌ Wrong";
    options.forEach(opt => {
      if (opt.getAttribute("onclick")?.includes("true")) {
        opt.classList.add('correct');
      }
    });
  }
}
